  <files>

  </files>
  <action>
    Create type assertion test file using Vitest 4.x expectTypeOf.
    
    Test structure:
    ```typescript
    import { defineEngineConfig } from '@pikacss/core'
    import { typography } from '../src'
    import { describe, expectTypeOf, it } from 'vitest'
    
    describe('plugin-typography types', () => {
      it('should augment EngineConfig with typography property', () => {
        const config = defineEngineConfig({
          plugins: [typography()],
          typography: {
            variables: {
              '--pk-prose-color-body': '#333',
              '--pk-prose-color-headings': '#111'
            }
          }
        })
        
        expectTypeOf(config).toHaveProperty('typography')
        expectTypeOf(config.typography).toMatchTypeOf<{
          variables?: Partial<Record<string, string>>
        } | undefined>()
      })
      
      it('should accept valid CSS variable names', () => {
        const config = {
          plugins: [typography()],
          typography: {
            variables: {
              '--pk-prose-color-body': '#333',
              '--custom-var': 'value'  // Any CSS variable allowed
            }
          }
        }
        expectTypeOf(config).toMatchTypeOf(defineEngineConfig)
      })
      
      it('should work without typography config', () => {
        const config = defineEngineConfig({
          plugins: [typography()]
        })
        expectTypeOf(config).toMatchTypeOf<{ plugins: any[] }>()
      })
      
      it('should enforce correct variable value types', () => {
        const config = {
          plugins: [typography()],
          typography: {
            variables: {
              // @ts-expect-error - numbers not allowed, only strings
              '--pk-prose-color-body': 123
            }
          }
        }
        expectTypeOf(config).not.toMatchTypeOf(defineEngineConfig)
      })
    })
    ```
    
    These tests verify:
    1. Module augmentation loads typography property
    2. Variables object accepts string values
    3. Typography config is optional
    4. Type safety for variable values
  </action>
