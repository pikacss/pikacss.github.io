  <action>
    Create type assertion test file using Vitest 4.x expectTypeOf, testing all configuration options.
    
    Test structure:
    ```typescript
    import { defineEngineConfig } from '@pikacss/core'
    import { icons } from '../src'
    import { describe, expectTypeOf, it } from 'vitest'
    
    describe('plugin-icons types', () => {
      it('should augment EngineConfig with icons property', () => {
        const config = defineEngineConfig({
          plugins: [icons()],
          icons: {
            prefix: 'icon-',
            scale: 1.2,
            mode: 'mask'
          }
        })
        
        expectTypeOf(config).toHaveProperty('icons')
        expectTypeOf(config.icons).toMatchTypeOf<{
          prefix?: string
          scale?: number
          mode?: 'auto' | 'mask' | 'bg'
        } | undefined>()
      })
      
      it('should enforce valid mode values', () => {
        const validConfig = {
          plugins: [icons()],
          icons: {
            mode: 'auto' as const  // Valid
          }
        }
        expectTypeOf(validConfig).toMatchTypeOf(defineEngineConfig)
        
        const invalidConfig = {
          plugins: [icons()],
          icons: {
            // @ts-expect-error - invalid mode value
            mode: 'invalid'
          }
        }
        expectTypeOf(invalidConfig).not.toMatchTypeOf(defineEngineConfig)
      })
      
      it('should accept partial configuration', () => {
        // Only prefix
        const config1 = defineEngineConfig({
          plugins: [icons()],
          icons: { prefix: 'i-' }
        })
        expectTypeOf(config1.icons).toMatchTypeOf<{ prefix?: string } | undefined>()
        
        // Only scale
        const config2 = defineEngineConfig({
          plugins: [icons()],
          icons: { scale: 1.5 }
        })
        expectTypeOf(config2.icons).toMatchTypeOf<{ scale?: number } | undefined>()
        
        // Only mode
        const config3 = defineEngineConfig({
          plugins: [icons()],
          icons: { mode: 'bg' }
        })
        expectTypeOf(config3.icons).toMatchTypeOf<{ mode?: 'auto' | 'mask' | 'bg' } | undefined>()
      })
      
      it('should work without icons config', () => {
        const config = defineEngineConfig({
          plugins: [icons()]
        })
        expectTypeOf(config).toMatchTypeOf<{ plugins: any[] }>()
      })
      
      it('should enforce correct option types', () => {
        const config = {
          plugins: [icons()],
          icons: {
            // @ts-expect-error - prefix must be string
            prefix: 123,
            // @ts-expect-error - scale must be number
            scale: 'large'
          }
        }
        expectTypeOf(config).not.toMatchTypeOf(defineEngineConfig)
      })
    })
    ```
    
    These tests verify:
    1. All three configuration options (prefix, scale, mode)
    2. Mode enum values enforced
    3. Partial configuration supported
    4. Type safety for all options
