   - Check plugin examples reference actual plugin APIs
   - Validate TypeScript types used in examples

**Implementation Pattern:**
```typescript
import { describe, expect, it } from 'vitest'
import { readFileSync } from 'node:fs'
import { extractPackageAPIs } from '../../src/core/extractor.js'
import { getMonorepoPackages } from '../../src/utils/package.js'

describe('pikacss-expert/SKILL.md API Validation', () => {
  const skillContent = readFileSync('.github/skills/pikacss-expert/SKILL.md', 'utf-8')
  
  it('all API examples reference verified exports', async () => {
    // Extract API calls from code blocks (simple heuristic)
    const codeBlocks = skillContent.match(/```(?:ts|typescript)\n([\s\S]*?)```/g) || []
    const apiCalls = new Set<string>()
    
    codeBlocks.forEach(block => {
      // Match function calls like createEngine(, defineEngineConfig(
      const calls = block.match(/\b(create\w+|define\w+|use\w+)\(/g) || []
      calls.forEach(call => apiCalls.add(call.replace('(', '')))
    })
    
    // Get actual APIs from packages
    const packages = getMonorepoPackages()
    const coreAPIs = await extractPackageAPIs(packages.find(p => p.name === '@pikacss/core')!)
    const functionNames = coreAPIs.exports
      .filter(e => e.kind === 'function')
      .map(e => e.name)
    
    // Verify documented APIs exist
    apiCalls.forEach(api => {
      expect(functionNames).toContain(api, 
        `API ${api} used in examples but not exported from @pikacss/core`)
    })
  })
  
  it('import paths match package exports', () => {
    // Extract import statements
    const imports = [...skillContent.matchAll(/import\s+{([^}]+)}\s+from\s+['"]([^'"]+)['"]/g)]
    
    imports.forEach(([_, names, pkg]) => {
      if (pkg.startsWith('@pikacss/')) {
        // Verify package exists
        const packageJsonPath = pkg.replace('@pikacss/', 'packages/') + '/package.json'
        expect(() => readFileSync(packageJsonPath, 'utf-8')).not.toThrow()
      }
    })
  })
})
```
