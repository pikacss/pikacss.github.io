```typescript
import { beforeEach, describe, expect, it } from 'vitest'
import { createCtx } from '../../src/ctx'

describe('Edge Cases', () => {
  let ctx

  beforeEach(async () => {
    ctx = createCtx({
      cwd: '/mock/project',
      currentPackageName: '@pikacss/integration',
      scan: { include: ['**/*.ts'], exclude: ['node_modules/**'] },
      configOrPath: null,
      fnName: 'pika',
      transformedFormat: 'string' as const,
      tsCodegen: false,
      cssCodegen: 'pika.gen.css',
      autoCreateConfig: false
    })
    await ctx.setup()
  })

  describe('Malformed Code', () => {
    it('should handle syntax errors in pika() arguments', async () => {
      const code = 'const s = pika({ color: })'
      
      await expect(
        ctx.transform(code, 'bad.ts')
      ).rejects.toThrow()
    })

    it('should handle missing closing brace', async () => {
      const code = 'const s = pika({ color: "red"'
      
      await expect(
        ctx.transform(code, 'bad.ts')
      ).rejects.toThrow()
    })

    it('should handle invalid object syntax', async () => {
      const code = 'const s = pika({ : "red" })'
      
      await expect(
        ctx.transform(code, 'bad.ts')
      ).rejects.toThrow()
    })
  })

  describe('Comments and Strings', () => {
    it('should ignore pika() in line comments', async () => {
      const code = `
        // const fake = pika({ color: 'red' })
        const real = pika({ color: 'blue' })
      `
      const result = await ctx.transform(code, 'test.ts')
      const css = await ctx.getCssCodegenContent()
      
      expect(css).not.toContain('red')
      expect(css).toContain('blue')
    })

    it('should ignore pika() in block comments', async () => {
      const code = `
        /* const fake = pika({ color: 'red' }) */
        const real = pika({ color: 'blue' })
      `
      const result = await ctx.transform(code, 'test.ts')
      const css = await ctx.getCssCodegenContent()
      
      expect(css).not.toContain('red')
      expect(css).toContain('blue')
    })

    it('should ignore pika() in string literals', async () => {
      const code = `
        const str = "const fake = pika({ color: 'red' })"
        const real = pika({ color: 'blue' })
      `
      const result = await ctx.transform(code, 'test.ts')
      const css = await ctx.getCssCodegenContent()
      
      expect(css).not.toContain('red')
      expect(css).toContain('blue')
    })
  })

  describe('Nested and Invalid Calls', () => {
    it('should handle nested pika() calls', async () => {
      const code = 'const s = pika({ color: pika({ color: "red" }) })'
      
      // Should either error or handle gracefully
      await expect(
        ctx.transform(code, 'nested.ts')
      ).rejects.toThrow()
    })

    it('should handle empty pika() call', async () => {
      const code = 'const s = pika()'
      
      const result = await ctx.transform(code, 'empty.ts')
      // Should handle gracefully or error
      expect(result).toBeDefined()
    })

    it('should handle pika() with no arguments', async () => {
      const code = 'const s = pika({})'
      
      const result = await ctx.transform(code, 'empty.ts')
      expect(result).toBeDefined()
      expect(result.code).toBeDefined()
    })
  })

  describe('Build-time Constraint Violations', () => {
    it('should detect undefined variable references in build-time scope', async () => {
      const code = `
        function Component({ color }) {
          const styles = pika({ color })
        }
      `
      
      // Build-time evaluation with new Function() should fail on undefined 'color' in eval scope
      await expect(
        ctx.transform(code, 'scope-error.ts')
      ).rejects.toThrow()
    })

    it('should allow constants in pika() calls', async () => {
      const code = `
        const COLOR = 'red'
        const styles = pika({ color: COLOR })
      `
      
      // Constants are statically analyzable - should work
      const result = await ctx.transform(code, 'const.ts')
      expect(result).toBeDefined()
      
      const css = await ctx.getCssCodegenContent()
      expect(css).toContain('color:red')
    })
  })

  describe('Engine State', () => {
    it('should handle missing engine gracefully', async () => {
      const ctxNoSetup = createCtx({
        cwd: '/mock/project',
        currentPackageName: '@pikacss/integration',
        scan: { include: ['**/*.ts'], exclude: ['node_modules/**'] },
        configOrPath: null,
        fnName: 'pika',
        transformedFormat: 'string' as const,
        tsCodegen: false,
        cssCodegen: 'pika.gen.css',
        autoCreateConfig: false
      })
      // Don't call setup() - engine is null
      
      const result = await ctxNoSetup.transform('const s = pika({ color: "red" })', 'test.ts')
      expect(result).toBeNull() // Gracefully returns null
    })

    it('should handle invalid CSS properties', async () => {
      const code = 'const s = pika({ invalidProperty: "value" })'
      
      // Core engine should handle or reject invalid properties
      const result = await ctx.transform(code, 'invalid.ts')
      expect(result).toBeDefined()
    })
  })

  describe('Config Loading Errors', () => {
    it('should handle missing config file when not auto-creating', async () => {
      const ctxNoConfig = createCtx({
        cwd: '/nonexistent/path',
        currentPackageName: '@pikacss/integration',
        scan: { include: ['**/*.ts'], exclude: ['node_modules/**'] },
        configOrPath: 'nonexistent.config.js',
        fnName: 'pika',
        transformedFormat: 'string' as const,
        tsCodegen: false,
        cssCodegen: 'pika.gen.css',
        autoCreateConfig: false
      })
      
      // Should handle missing config gracefully
      await ctxNoConfig.setup()
      expect(ctxNoConfig.resolvedConfig).toBeDefined()
    })
  })

  describe('Null/Undefined Handling', () => {
    it('should handle null code', async () => {
      await expect(
        ctx.transform(null as any, 'test.ts')
      ).rejects.toThrow()
    })

    it('should handle undefined code', async () => {
      await expect(
        ctx.transform(undefined as any, 'test.ts')
      ).rejects.toThrow()
    })

    it('should handle empty string code', async () => {
      const result = await ctx.transform('', 'empty.ts')
      expect(result).toBeDefined()
    })
  })
})
```
